 65    def _is_pycdlib_dir(
 66        self,
 67        record,  #: pycdlib.DirectoryRecord
 68    ) -> bool:

 72    def _is_pycdlib_symlink(
 73        self,
 74        record,  #: pycdlib.pycdlib.backends.pycdlibstructures.DirectoryRecord
 75    ) -> bool:
 76        """Checks if a pycdlib DirectoryRecord is a symbolic link (using Rock Ridge)."""
 77        # pycdlib doesn't directly expose a simple is_symlink().
 78        # We need to check for Rock Ridge 'SL' extension.
 79        if record.has_rock_ridge():
 80            rr_extensions = record.rock_ridge_extensions()
 81            for ext in rr_extensions:
 82                if isinstance(ext, pycdlib.rockridge.SLExtension):
 83                    return True
 84        return False

 86    def _convert_direntry_to_member(
 87        self,
 88        path: str,
 89        record,  #: pycdlib.backends.pycdlibstructures.DirectoryRecord
 90    ) -> ArchiveMember:
 91        """Converts a pycdlib DirectoryRecord to an ArchiveMember."""
 92        member_type = MemberType.OTHER
 93        if self._is_pycdlib_dir(record):
 94            member_type = MemberType.DIR
 95        elif self._is_pycdlib_symlink(record):  # Check for symlink before regular file
 96            member_type = MemberType.LINK
 97        elif record.is_file():
 98            member_type = MemberType.FILE
 99
100        mtime_dt: Optional[datetime] = None
101        if record.date_time:
102            try:
103                # pycdlib's datetime is timezone-naive. Assume UTC or local as per ISO standard context.
104                # Python's datetime.fromtimestamp will use local timezone.
105                # For consistency, one might want to specify UTC if the ISO timestamps are UTC.
106                # However, pycdlib already gives a datetime object.
107                mtime_dt = record.date_time
108            except ValueError:  # Should not happen if record.date_time is valid
109                mtime_dt = None
110
111        link_target = None
112        if member_type == MemberType.LINK and record.has_rock_ridge():
113            rr_extensions = record.rock_ridge_extensions()
114            for ext in rr_extensions:
115                if isinstance(ext, pycdlib.rockridge.SLExtension):
116                    # SLExtension stores components of the link target.
117                    # We need to join them. pycdlib doesn't seem to have a direct
118                    # method to get the full link target path string easily.
119                    # This is a simplified assumption; complex SL links might exist.
120                    link_target = "/".join(
121                        c.data.decode("utf-8", "replace") for c in ext.sl_components
122                    )
123                    break
124
125        # pycdlib does not directly expose CRC32 or compression method for individual files
126        # as ISO9660 is not a compressed format in the same way as zip/rar.
127        return ArchiveMember(
128            filename=path,
129            file_size=record.data_length
130            if record.is_file() or self._is_pycdlib_symlink(record)
131            else 0,  # Symlinks also have a size for their target path
132            compress_size=record.data_length
133            if record.is_file() or self._is_pycdlib_symlink(record)
134            else 0,  # No compression in ISO
135            mtime=mtime_dt,
136            type=member_type,
137            mode=record.posix_mode()
138            if record.has_rock_ridge() and hasattr(record, "posix_mode")
139            else None,
140            link_target=link_target,
141            raw_info=record,
142        )

144    def _walk_iso(
145        self, current_path: str = "/", current_iso_path: str = "/"
146    ) -> Iterator[ArchiveMember]:
147        """Helper to recursively walk the ISO directory structure."""
148        if not self.iso:
149            raise ArchiveError("ISO not opened")
150
151        try:
152            for child_name_bytes, record in self.iso.list_dir(
153                iso_path=current_iso_path
154            ):
155                child_name = child_name_bytes.decode(
156                    self.iso.joliet_output_encoding
157                    if self.iso.has_joliet()
158                    else self.iso.iso_output_encoding,
159                    "replace",
160                )
161
162                # Skip '.' and '..' entries
163                if child_name == "." or child_name == "..":
164                    continue
165
166                # Construct the full path for the member
167                # Ensure no double slashes if current_path is '/'
168                member_path = os.path.join(current_path.strip("/"), child_name).replace(
169                    "\\", "/"
170                )
171                if (
172                    not member_path.startswith("/") and current_path == "/"
173                ):  # Handle root children
174                    member_path = f"/{member_path}"
175                if current_path != "/" and not member_path.startswith(
176                    current_path
177                ):  # ensure children of /foo are /foo/bar
178                    member_path = f"{current_path.rstrip('/')}/{child_name}".replace(
179                        "\\", "/"
180                    )
181
182                # Clean path: remove leading slash for internal consistency if desired,
183                # but usually archive members are listed with full paths from root.
184                # For now, keep leading slash.
185                # member_path = member_path.lstrip('/') # Example if no leading slash is wanted
186
187                yield self._convert_direntry_to_member(member_path, record)
188
189                if self._is_pycdlib_dir(record):
190                    # Construct the iso_path for pycdlib (needs trailing slash for dirs)
191                    next_iso_path = (
192                        os.path.join(current_iso_path, child_name).replace("\\", "/")
193                        + "/"
194                    )
195                    yield from self._walk_iso(member_path, next_iso_path)
196        except PyCdlibException as e:
197            raise ArchiveError(
198                f"Error walking ISO directory {current_iso_path}: {e}"
199            ) from e

206    def iter_members_with_io(
207        self,
208        filter: Callable[[ArchiveMember], bool] | None = None,
209        *,
210        pwd: bytes | str | None = None,
211    ) -> Iterator[tuple[ArchiveMember, IO[bytes] | None]]:
212        if not self.iso:
213            raise ArchiveError("ISO not opened")
214
215        if pwd is not None:
216            raise ArchiveError("Password is not supported for ISOReader")
217
218        # Yield the root directory itself, as it's a common convention
219        # pycdlib doesn't explicitly list the root dir record in list_dir('/')
220        # We can try to get its record specifically.
221        try:
222            root_record = self.iso.get_record(iso_path="/.")
223            if root_record:
224                yield self._convert_direntry_to_member("/", root_record)
225        except PyCdlibException:
226            # This might fail if the ISO is very strange or empty.
227            # Log or handle as appropriate. For now, proceed to walk.
228            pass
229
230        for member in self._walk_iso(current_path="/", current_iso_path="/"):
231            if filter is None or filter(member):
232                try:
233                    stream = self.open(member)
234                    yield member, stream
235                    stream.close()
236                except (ArchiveError, OSError) as e:
237                    logger.info(f"Error opening member {member.filename}: {e}")

239    def open(
240        self, member: ArchiveMember | str, *, pwd: Optional[str | bytes] = None
241    ) -> IO[bytes]:
242        if not self.iso:
243            raise ArchiveError("ISO not opened")
244
245        member_name: str
246        if isinstance(member, ArchiveMember):
247            member_name = member.filename
248        elif isinstance(member, str):
249            member_name = member
250        else:
251            raise TypeError("member must be an ArchiveMember or a string path")
252
253        # pycdlib needs the exact path as it's stored in the ISO
254        # Our member.filename should correspond to this.
255        # Ensure it's in the format pycdlib expects (e.g. /FOO/BAR.TXT;1)
256        # The _walk_iso method should produce these correct paths.
257        # If the member name includes a version (';1'), pycdlib handles it.
258        # If it doesn't, pycdlib usually defaults to the first version.
259
260        # Convert our potentially clean path to the ISO path format pycdlib expects
261        # This means it must start with '/' and use '/' as separator.
262        iso_path = member_name
263        if not iso_path.startswith("/"):
264            iso_path = "/" + iso_path
265
266        # Ensure the path uses '/' as separator, which pycdlib expects.
267        iso_path = iso_path.replace("\\", "/")
268
269        try:
270            # Check if it's a directory, pycdlib can't extract directories as a stream
271            record = self.iso.get_record(iso_path=iso_path)
272            if record and self._is_pycdlib_dir(record):
273                raise IsADirectoryError(
274                    f"Cannot open directory '{member_name}' as a file stream."
275                )
276
277            file_data = self.iso.get_file_from_iso(iso_path=iso_path)
278            return io.BytesIO(file_data)
279        except pycdlib.backends.pycdlibexceptions.PyCdlibInvalidInput:
280            # This exception is often raised for "file not found"
281            raise FileNotFoundError(f"File not found in ISO: {member_name}") from None
282        except PyCdlibException as e:
283            raise ArchiveError(
284                f"Failed to open member '{member_name}' from ISO: {e}"
285            ) from e

287    def get_archive_info(self) -> ArchiveInfo:
288        if not self.iso:
289            raise ArchiveError("ISO not opened")
290
291        # Basic info
292        info = ArchiveInfo(format=self.format.value)
293
294        # Try to get volume label
295        # pycdlib might have this on the primary volume descriptor
296        try:
297            pvd = self.iso.pvd
298            if pvd:
299                info.comment = pvd.volume_identifier.decode(
300                    self.iso.iso_output_encoding, "replace"
301                ).strip()
302                # Other PVD fields like system_identifier, volume_set_identifier,
303                # publisher_identifier, data_preparer_identifier, application_identifier
304                # could be added to extra_info if needed.
305                extra = {}
306                if pvd.system_identifier:
307                    extra["system_identifier"] = pvd.system_identifier.decode(
308                        self.iso.iso_output_encoding, "replace"
309                    ).strip()
310                if pvd.publisher_identifier:
311                    extra["publisher_identifier"] = pvd.publisher_identifier.decode(
312                        self.iso.iso_output_encoding, "replace"
313                    ).strip()
314                if pvd.application_identifier:
315                    extra["application_identifier"] = pvd.application_identifier.decode(
316                        self.iso.iso_output_encoding, "replace"
317                    ).strip()
318                if extra:
319                    info.extra = extra
320        except PyCdlibException:
321            # Silently ignore if PVD info is not available or fails to parse
322            pass
323
324        # ISOs are not inherently solid, encrypted (standardly), or versioned like RAR/7z
325        info.is_solid = False
326        # info.version could perhaps indicate ISO9660 version (e.g., "1"), Joliet, Rock Ridge presence
327        # For now, keeping it simple.
328
329        return info

26    path_or_file: str | bytes | IO[bytes],
27) -> ArchiveFormat:
28    # [signature, ...], offset, format
29    SIGNATURES = [
30        ([b"\x50\x4b\x03\x04"], 0, ArchiveFormat.ZIP),
31        (
32            [
33                b"\x52\x61\x72\x21\x1a\x07\x00",  # RAR4
34                b"\x52\x61\x72\x21\x1a\x07\x01\x00",  # RAR5
35            ],
36            0,
37            ArchiveFormat.RAR,
38        ),
39        ([b"\x37\x7a\xbc\xaf\x27\x1c"], 0, ArchiveFormat.SEVENZIP),
40        ([b"\x1f\x8b"], 0, ArchiveFormat.GZIP),
41        ([b"\x42\x5a\x68"], 0, ArchiveFormat.BZIP2),
42        ([b"\xfd\x37\x7a\x58\x5a\x00"], 0, ArchiveFormat.XZ),
43        ([b"\x28\xb5\x2f\xfd"], 0, ArchiveFormat.ZSTD),
44        ([b"\x04\x22\x4d\x18"], 0, ArchiveFormat.LZ4),
45        ([b"ustar"], 257, ArchiveFormat.TAR),  # TAR "ustar" magic
46        (_ISO_MAGIC_BYTES, 0x8001, ArchiveFormat.ISO),  # ISO9660
47    ]
48    f: IO[bytes]
49
50    if isinstance(path_or_file, (str, bytes)):
51        # If it's a path, check if it's a directory first
52        if os.path.isdir(path_or_file):
53            return ArchiveFormat.FOLDER
54        try:
55            f = open(path_or_file, "rb")
56            close_after = True
57        except FileNotFoundError:
58            return (
59                ArchiveFormat.UNKNOWN
60            )  # Or raise error, depending on desired behavior
61    elif hasattr(path_or_file, "read") and hasattr(path_or_file, "seek"):
62        f = cast(IO[bytes], path_or_file)
63        # We can't check is_dir on a stream, assume it's not a folder for streams
64        close_after = False
65    else:
66        # Not a path and not a stream
67        return ArchiveFormat.UNKNOWN
68
69    try:
70        for magics, offset, fmt in SIGNATURES:
71            bytes_to_read = max(len(magic) for magic in magics)
72            f.seek(offset)
73            data = f.read(bytes_to_read)
74            if any(data.startswith(magic) for magic in magics):
75                return fmt
76
77    finally:
78        if close_after:
79            f.close()
